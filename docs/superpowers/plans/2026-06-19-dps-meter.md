function meterChart(hostId, rows, unit, footerLabel) {
  rows = rows.slice().sort((a,b)=>b.value-a.value);
  const total = rows.reduce((s,r)=>s+r.value, 0);
  const maxVal = rows.length ? rows[0].value : 1;
  let html = '<div class="meter-list">';
  rows.forEach(r => {
    const pct = total > 0 ? Math.round(r.value / total * 100) : 0;
    const barW = maxVal > 0 ? Math.round(r.value / maxVal * 100) : 0;
    html += `<div class="meter-row-grid">
      <span class="meter-label">${r.name}</span>
      <div class="meter-bar-wrap"><div class="meter-bar-fill" style="width:${barW}%;background:${r.color}"></div></div>
      <span class="meter-value">${r.value.toFixed(1)} ${unit}</span>
      <span class="meter-pct">${pct}%</span>
    </div>`;
  });
  html += '</div>';
  if(footerLabel !== undefined) {
    html += `<div class="meter-footer">${footerLabel}: ${total.toFixed(1)} ${unit}</div>`;
  }
  $(hostId).innerHTML = html;
}
```

function pieChart(hostId, legendHostId, slices) {
  const total = slices.reduce((s,sl)=>s+sl.value, 0);
  if(total <= 0) { $(hostId).innerHTML = '<div class="empty" style="padding:20px;text-align:center">No damage dealt.</div>'; return; }
  const CX=80, CY=80, R=68, W=160, H=160;
  let svg = `<svg viewBox="0 0 ${W} ${H}" width="160" height="160" style="display:block;margin:0 auto">`;
  let startAngle = -Math.PI/2;
  slices.forEach(sl => {
    if(sl.value <= 0) return;
    const angle = (sl.value / total) * 2 * Math.PI;
    const endAngle = startAngle + angle;
    const x1 = CX + R * Math.cos(startAngle);
    const y1 = CY + R * Math.sin(startAngle);
    const x2 = CX + R * Math.cos(endAngle);
    const y2 = CY + R * Math.sin(endAngle);
    const large = angle > Math.PI ? 1 : 0;
    svg += `<path d="M${CX},${CY} L${x1.toFixed(2)},${y1.toFixed(2)} A${R},${R} 0 ${large},1 ${x2.toFixed(2)},${y2.toFixed(2)} Z" fill="${sl.color}" stroke="#efe6d2" stroke-width="1.5"/>`;
    // label at midpoint
    const mid = startAngle + angle/2;
    const lx = CX + R*0.62*Math.cos(mid);
    const ly = CY + R*0.62*Math.sin(mid);
    const pct = Math.round(sl.value/total*100);
    if(pct >= 5) svg += `<text x="${lx.toFixed(1)}" y="${(ly+4).toFixed(1)}" text-anchor="middle" style="font-size:11px;fill:#fff;font-weight:700;font-family:Georgia,serif">${pct}%</text>`;
    startAngle = endAngle;
  });
  svg += '</svg>';
  $(hostId).innerHTML = svg;
  if(legendHostId) {
    $(legendHostId).innerHTML = '<div class="pie-legend">' +
      slices.map(sl=>`<span><span class="pie-swatch" style="background:${sl.color}"></span>${sl.name}: ${sl.value.toFixed(1)} dps (${Math.round(sl.value/total*100)}%)</span>`).join('') +
      '</div>';
  }
}
```

      {name:"Cael (Keeper)", color:"#5b7f63", value:avgHeal},
    ], "hp/s", undefined);
    $("ch_keeperheal").innerHTML += `<div class="meter-footer">Avg HP/s restored across the fight</div>`;
  } else {
    $("ch_keeperheal").innerHTML='<div class="empty" style="padding:24px 14px;text-align:center">No rescues fired this engagement — Keeper spent all time on damage.</div>';
  }

  // Damage by type — pie chart
  const physDps = avgBy("warden") + avgBy("hunter") + series.reduce((s,f)=>s+(f.effBy?f.effBy.captain_phys||0:0),0)/series.length;
  const magicDps = series.reduce((s,f)=>s+(f.effBy?f.effBy.captain_magic||0:0),0)/series.length + avgBy("keeper");
  pieChart("ch_damagetype", "lg_damagetype", [
    {name:"Physical", color:"#c0392b", value:physDps},
    {name:"Magic",    color:"#7d5a93", value:magicDps},
  ]);

  // Armor mitigation — bar meter
  const avgPhysAbsorbed = avg("physMitTotal");
  const avgPhysThrough = physDps; // physical that got through is already in effBy
  if(P.phys > 0 && (avgPhysAbsorbed > 0 || avgPhysThrough > 0)) {
    meterChart("ch_armormet", [
      {name:"Absorbed by armor", color:"#c0392b", value:avgPhysAbsorbed},
      {name:"Through armor",     color:"#5b7f63", value:avgPhysThrough},
    ], "dps", "Total physical attempted");
  } else {
    $("ch_armormet").innerHTML='<div class="empty" style="padding:24px 14px;text-align:center">No armor in this fight — set Physical mit. % on the Setup tab to see the breakdown.</div>';
  }

  // Dread mitigation — bar meter
  const avgWillSaved = avg("willSaved");
  const avgHopeSaved = avg("hopeSaved");
  const avgDreadLost = series.reduce((s,f)=>s+(f.raw||0)*(f.dreadAfter!=null?0:0),0)/series.length; // computed below
  const avgDreadNet = series.reduce((s,f)=>{
    // dread cost = what we would have had without any dread, minus what we got
    // = raw*(1-physAfter) - eff (already net of phys and dread)
    return s + Math.max(0, (f.raw||0)*0.5 - 0); // approximation: use willSaved+hopeSaved as proxy
  },0)/series.length;
  const anyDread=series.some(s=>s.willSaved>0||s.hopeSaved>0);
  if(anyDread){
    const dreadLost=series.reduce((s,f)=>s+Math.max(0,(f.penSaved||0)+(f.willSaved||0)+(f.hopeSaved||0)-(f.dps||0)),0)/series.length;
    meterChart("ch_dreadmet", [
      {name:"Saved by Captain Will + Red Dawn", color:"#b8843c", value:avgWillSaved},
      {name:"Saved by Hope",                   color:"#c9a84c", value:avgHopeSaved},
    ], "dps", "Total Dread countered");
  } else {
    $("ch_dreadmet").innerHTML='<div class="empty" style="padding:24px 14px;text-align:center">No Dread in this fight — toggle Dread on the enemy to see how Will and Hope counter it.</div>';
  }

  // Wounds (unchanged line chart)
  lineChart("ch_wounds","lg_wounds",ORDER.map(k=>({name:M[k].tpl.name,color:TPL[k==="hunter"?"hunterA":k].color,pts:series.map(s=>({x:s.t,y:s.wounds[k]})),fmt:v=>Math.round(v)+" wounds"})),xmax,GRIEVOUS,{yfmt:v=>Math.round(v)});

  // Shadow / spirit (unchanged line chart, conditional)
  const sawShadow=series.some(s=>s.spirit.wil<s.spiritBase.wil-0.5||s.spirit.fat<s.spiritBase.fat-0.5);
  if(sawShadow){const spMax=Math.max(...series.map(s=>Math.max(s.spiritBase.wil,s.spiritBase.fat)));
    lineChart("ch_spirit","lg_spirit",[
      {name:"Party Will",color:"#7d5a93",pts:series.map(s=>({x:s.t,y:s.spirit.wil})),fmt:v=>Math.round(v)},
      {name:"Party Fate",color:"#b8843c",pts:series.map(s=>({x:s.t,y:s.spirit.fat})),fmt:v=>Math.round(v)},
    ],xmax,spMax,{});}
  else{$("ch_spirit").innerHTML='<div class="empty" style="padding:30px 14px;text-align:center">No Shadow in this fight — toggle Shadow on the enemy to watch it drain the party\'s Will and Fate (and Radiance hold it back).</div>';$("lg_spirit").innerHTML="";}

  // Inspiration timeline (unchanged)
  const insps=events.filter(e=>/Grace|Horn|Red Dawn|Black Arrow/.test(e.kind));
  $("insp_list").innerHTML=insps.length?insps.map(e=>`<div class="insp-row"><span class="it">${fmtT(e.t)}</span><span class="ik">${e.kind}</span><span>&#8212; ${e.who}</span></div>`).join(""):'<div class="empty">No Inspirations fired this engagement. They are rare grace — most fights pass without them.</div>';
}
