## Key Constants

**Recipe tier boundaries:** Apprentice = Lv 1–5, Journeyman = Lv 6–10, Craftsman = Lv 11+

**levelRequired per recipe:**
- hearthbread, wanderers_supper, contemplative_tea, rangers_fare, ember_porridge → 1
- bilberry_tea → 3
- iron_stew, stormcap_saute, moonpetal_tisane, heartflame_broth → 6
- restorative_broth → 8
- athelas_infusion → 12

**Ingredient rarity:** cultivated farm ingredients + common wild (moonpetal, ironbark_resin, stormcap, wanderer_fig, heartflame_pepper) → "common"; cave_truffle, bilberry, athelas → "uncommon"; seeds found during forage → "bonus"

**Mission vitalityRequired:** easy=0, medium=3, hard=6

**Mission success formula (no buffType):**
```
baseChance: easy=0.70, medium=0.45, hard=0.20
strengthRatio = min(1.0, foodStrength / requiredBuffStrength)
strengthBonus = strengthRatio * 0.25
successChance = min(0.95, baseChance + strengthBonus)
```

**Member starting stats (Might, Agility, Vitality, Will, Fate):**
- Aelindra: 2,2,4,5,3 | Thornwick: 2,3,3,4,5 | Old Mossback: 4,1,5,3,2
- Borin: 5,2,5,3,2 | Dagra: 4,4,4,3,3 | Keldra: 3,4,3,3,5 | Snorri: 4,2,5,4,1
- Reva: 3,5,3,4,4 | Silas: 2,4,3,5,3 | Marta: 5,3,5,2,3
- Aldric: 4,3,5,4,2 | Mira: 3,5,3,4,4 | Cael: 2,3,3,5,5

**Stat grants per mission:** success → +1 vitality +1 might per alive member; failure → +1 vitality only

**Seed shop prices:** all seeds 5g each. Sold: hearthgrain_seed, ironroot_seed, sunpetal_herb_seed, duskberry_seed, pale_cap_seed (no forest_honey_seed — honey comes from tended hives, not planted seeds)

```kotlin
data class RecipeTier(val label: String, val minLevel: Int, val recipes: List<Recipe>)

val tieredRecipes: StateFlow<List<RecipeTier>> = combine(
    inventory.observeIngredients(),
    player.observe()
) { items, playerState ->
    val cookingLevel = playerState?.cookingLevel ?: 1
    val qtyMap = items.associate { it.ingredientId to it.quantity }
    val tiers = listOf(
        RecipeTier("Apprentice", 1, gameData.recipes.filter { it.levelRequired <= 5 }),
        RecipeTier("Journeyman", 6, gameData.recipes.filter { it.levelRequired in 6..10 }),
        RecipeTier("Craftsman", 11, gameData.recipes.filter { it.levelRequired >= 11 })
    )
    tiers.map { tier ->
        val sorted = tier.recipes.sortedWith(
            compareByDescending<Recipe> { recipe ->
                recipe.ingredients.all { needed -> (qtyMap[needed.id] ?: 0) >= needed.qty }
            }.thenBy { it.levelRequired }
        )
        tier.copy(recipes = sorted)
    }
}.stateIn(viewModelScope, SharingStarted.WhileSubscribed(5000), emptyList())
```

override suspend fun doWork(): Result {
    val missionId = inputData.getString(KEY_MISSION_ID) ?: return Result.failure()
    val buffStrength = inputData.getInt(KEY_BUFF_STRENGTH, 0)

    val mission = gameData.missions.find { it.id == missionId } ?: return Result.failure()

    val baseChance = when (mission.difficulty) {
        "easy" -> 0.70f
        "medium" -> 0.45f
        "hard" -> 0.20f
        else -> 0.50f
    }
    val strengthRatio = if (mission.requiredBuffStrength > 0)
        minOf(1.0f, buffStrength.toFloat() / mission.requiredBuffStrength) else 1f
    val strengthBonus = strengthRatio * 0.25f
    val successChance = minOf(0.95f, baseChance + strengthBonus)

    val succeeded = Random.nextFloat() < successChance

    if (succeeded) {
        val multiplier = mission.rewardMultiplier
        val money = Random.nextInt(mission.rewardMoneyMin, mission.rewardMoneyMax + 1) * multiplier
        player.addMoney(money)
        val rewardCount = Random.nextInt(1, 4) + (multiplier - 1)
        mission.rewardTable.shuffled().take(rewardCount).forEach {
            inventory.addIngredient(it, 1)
        }
        notify("Mission Complete", "${mission.name} — your band has returned.", NOTIFICATION_ID)
    } else {
        applyFailureConsequences(mission.difficulty, mission.bandId, strengthRatio)
        notify("Mission Failed", buildFailureMessage(mission.name, mission.difficulty), NOTIFICATION_ID)
    }

    band.grantMissionStats(mission.bandId, succeeded)
    sessions.clearMission()
    return Result.success()
}

        "medium" -> {

package com.liquidcode7.hearthcraft.ui.viewmodel

import androidx.lifecycle.ViewModel
import androidx.lifecycle.viewModelScope
import com.liquidcode7.hearthcraft.data.repository.InventoryRepository
import com.liquidcode7.hearthcraft.data.repository.PlayerRepository
import dagger.hilt.android.lifecycle.HiltViewModel
import kotlinx.coroutines.flow.SharingStarted
import kotlinx.coroutines.flow.StateFlow
import kotlinx.coroutines.flow.map
import kotlinx.coroutines.flow.stateIn
import kotlinx.coroutines.launch
import javax.inject.Inject

data class SeedForSale(
    val seedId: String,
    val name: String,
    val priceGold: Int,
    val ownedQty: Int
)

@HiltViewModel
class MarketViewModel @Inject constructor(
    private val player: PlayerRepository,
    private val inventory: InventoryRepository
) : ViewModel() {

    val gold: StateFlow<Int> = player.observe()
        .map { it?.money ?: 0 }
        .stateIn(viewModelScope, SharingStarted.WhileSubscribed(5000), 0)

    val seedsForSale: StateFlow<List<SeedForSale>> = inventory.observeSeeds()
        .map { stocks ->
            val stockMap = stocks.associate { it.seedId to it.quantity }
            SEED_CATALOGUE.map { (seedId, name) ->
                SeedForSale(
                    seedId = seedId,
                    name = name,
                    priceGold = SEED_PRICE_GOLD,
                    ownedQty = stockMap[seedId] ?: 0
                )
            }
        }
        .stateIn(viewModelScope, SharingStarted.WhileSubscribed(5000), emptyList())

    fun buySeed(seedId: String) {
        viewModelScope.launch {
            val purchased = player.spendMoney(SEED_PRICE_GOLD)
            if (purchased) {
                inventory.addSeed(seedId, 1)
            }
        }
    }

    companion object {
        const val SEED_PRICE_GOLD = 5
        val SEED_CATALOGUE = listOf(
            "hearthgrain_seed" to "Hearthgrain Seed",
            "ironroot_seed" to "Ironroot Seed",
            "sunpetal_herb_seed" to "Sunpetal Herb Seed",
            "duskberry_seed" to "Duskberry Seed",
            "pale_cap_seed" to "Pale Cap Seed"
        )
    }
}
